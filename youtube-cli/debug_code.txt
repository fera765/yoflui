// Código de debug exemplo
console.log('Iniciando processo de debug...');
function debugFunction() {
    try {
        console.log('Executando função de debug');
        // Código de debug aqui
        const debugInfo = {
            timestamp: new Date().toISOString(),
            status: 'active',
            version: '1.0.0'
        };
        console.log('Informações de debug:', debugInfo);
        return debugInfo;
    } catch (error) {
        console.error('Erro no debug:', error);
        return null;
    }
}

debugFunction();
